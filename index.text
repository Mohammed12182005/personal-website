function simpleHash(str){
  let h = 5381;
  for (let i=0;i<str.length;i++){ h = ((h<<5)+h+str.charCodeAt(i)) >>> 0; }
  return h.toString(16);
}

function toEmbedUrl(url){
  try{
    if (url.includes("youtu.be/")) return "https://www.youtube.com/embed/" + url.split("youtu.be/")[1].split(/[?&]/)[0];
    if (url.includes("watch?v=")) return "https://www.youtube.com/embed/" + url.split("watch?v=")[1].split("&")[0];
    if (url.includes("vimeo.com/")) return "https://player.vimeo.com/video/" + url.split("vimeo.com/")[1].split(/[?&]/)[0];
  }catch(e){}
  return url;
}

function openProjectDetail(id){
  const p = projects.find(x => x.id === id);
  if (!p) return;
  document.getElementById("projectModalTitle").textContent = p.title;
  document.getElementById("projectModalTags").innerHTML = (p.tags||[]).map(t => `<span class="tag">${escapeHtml(t)}</span>`).join("");
  document.getElementById("projectModalDesc").textContent = p.description || p.summary;

  let mediaHtml;
  if (p.video){
    mediaHtml = /youtube\.com|youtu\.be|vimeo\.com/.test(p.video)
      ? `<iframe src="${escapeHtml(toEmbedUrl(p.video))}" allowfullscreen></iframe>`
      : `<video src="${escapeHtml(p.video)}" controls></video>`;
  } else if (p.image){
    mediaHtml = `<img src="${escapeHtml(p.image)}" alt="${escapeHtml(p.title)}">`;
  } else {
    mediaHtml = `<div class="modal-media-placeholder">No media yet</div>`;
  }
  document.getElementById("projectModalMedia").innerHTML = mediaHtml;

  let links = "";
  if (p.demoUrl) links += `<a class="btn btn-primary btn-sm" href="${escapeHtml(p.demoUrl)}" target="_blank" rel="noopener"><i class="fa-solid fa-arrow-up-right-from-square"></i> Live demo</a>`;
  if (p.repoUrl) links += `<a class="btn btn-outline btn-sm" href="${escapeHtml(p.repoUrl)}" target="_blank" rel="noopener"><i class="fa-brands fa-github"></i> Repository</a>`;
  document.getElementById("projectModalLinks").innerHTML = links;

  openModal("projectModal");
}

document.getElementById("projectForm").addEventListener("submit", (e) => {
  e.preventDefault();
  const id = document.getElementById("projectId").value || ("p" + Date.now());
  const fileInput = document.getElementById("pfImageFile");

  const save = (image) => {
    const data = {
      id,
      title: document.getElementById("pfTitle").value.trim(),
      summary: document.getElementById("pfSummary").value.trim(),
      description: document.getElementById("pfDescription").value.trim(),
      tags: document.getElementById("pfTags").value.split(",").map(t => t.trim()).filter(Boolean),
      image,
      video: document.getElementById("pfVideoUrl").value.trim(),
      demoUrl: document.getElementById("pfDemoUrl").value.trim(),
      repoUrl: document.getElementById("pfRepoUrl").value.trim()
    };
    const idx = projects.findIndex(p => p.id === id);
    if (idx > -1) projects[idx] = data; else projects.push(data);
    closeModal("projectFormModal");
    renderProjects();
    showToast(idx > -1 ? "Project updated." : "Project added.");
  };

  if (fileInput.files && fileInput.files[0]){
    const reader = new FileReader();
    reader.onload = () => save(reader.result);
    reader.readAsDataURL(fileInput.files[0]);
  } else {
    save(document.getElementById("pfImageUrl").value.trim() || e.target.dataset.existingImage || "");
  }
});

document.addEventListener("click", (e) => {
  const tabBtn = e.target.closest("[data-tab]");
  if (tabBtn){ switchTab(tabBtn.dataset.tab); }

  const closeBtn = e.target.closest("[data-close]");
  if (closeBtn){ closeModal(closeBtn.dataset.close); }

  if (e.target.closest("#hamburgerBtn")) document.getElementById("navLinks").classList.toggle("open");
  if (e.target.closest("#lockBtn")) { isOwner ? logout() : openModal("loginModal"); }

  const tagBtn = e.target.closest(".tag-filter");
  if (tagBtn){ activeTag = tagBtn.dataset.tag; renderProjects(); }

  if (e.target.closest("#addProjectBtn") || e.target.closest("#addProjectCard")) openProjectForm();

  const editBtn = e.target.closest("[data-edit]");
  if (editBtn){ openProjectForm(editBtn.dataset.edit); return; }
  const delBtn = e.target.closest("[data-delete]");
  if (delBtn){ deleteProject(delBtn.dataset.delete); return; }

  if (e.target.closest("#exportBtn")) exportProjects();
  if (e.target.closest("#resetBtn")) resetProjects();

  const card = e.target.closest(".project-plate");
  if (card) openProjectDetail(card.dataset.id);

  if (e.target.classList.contains("modal-overlay")) e.target.classList.remove("open");
});
